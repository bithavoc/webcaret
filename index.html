Example API App:

import webcaret.application;
import heaploop.looping;

void main() {
    loop ^^= {
        auto app = new Application;
        app.router.get("/") ^ (req, res) {
            res.write("Hello World");
            res.end;
        };
        app.serve("0.0.0.0", 3000);
    };
}
